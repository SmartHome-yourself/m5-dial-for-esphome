46 /*
47 * Converts an HUE to r, g or b.
48 * returns float in the set [0, 1].
49 */
50 float hue2rgb(float p, float q, float t) {
51 if (t < 0)
52 t += 1;
53 if (t > 1)
54 t -= 1;
55 if (t < 1./6)
56 return p + (q - p) * 6 * t;
57 if (t < 1./2)
58 return q;
59 if (t < 2./3)
60 return p + (q - p) * (2./3 - t) * 6;
61
62 return p;
63 }
64
65 /*
66 * Convert HSL to RGB.
67 * http://en.wikipedia.org/wiki/HSL_color_space.
68 *
69 * h, s, l [float 0 - 1]
70 *
71 * returns color888
72 */
73 uint32_t hslToRgb(float h, float s, float l) {
74 RGB result;
75
76 if(0 == s) {
77 result.r = result.g = result.b = l; // achromatic
78 }
79 else {
80 float q = l < 0.5 ? l * (1 + s) : l + s - l * s;
81 float p = 2 * l - q;
82 result.r = hue2rgb(p, q, h + 1./3) * 255;
83 result.g = hue2rgb(p, q, h) * 255;
84 result.b = hue2rgb(p, q, h - 1./3) * 255;
85 }
86
87 return M5Dial.Display.color888(result.r, result.g, result.b);
88 }

286 float getDegByCoord(uint16_t x, uint16_t y){
287 float mx = M5Dial.Display.width()/2;
288 float my = M5Dial.Display.height()/2;
289
290 float angle = atan2(y - my, x - mx) * 180.0 / M_PI;
291 angle = 360 - fmod((angle + 360.0 - 90), 360.0);
292
293 return angle;
294 }
